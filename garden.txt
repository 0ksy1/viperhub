-- === –õ–æ–≥–∏–∫–∞ –∫–Ω–æ–ø–∫–∏ START ===
local farming = false
local farmThread

startBtn.MouseButton1Click:Connect(function()
    if farming then
        farming = false
        addLog("‚õî –¶–∏–∫–ª –æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω –≤—Ä—É—á–Ω—É—é.")
        startBtn.Text = "START"
        return
    end

    farming = true
    startBtn.Text = "STOP"
    addLog("üöÄ –¶–∏–∫–ª —Ñ–∞—Ä–º–∞ –∑–∞–ø—É—â–µ–Ω.")

    farmThread = task.spawn(function()
        local unitProgress = {}
        local placedUnits = 0
        local currentLevel = nil
        local placedCentral = false
        local placedAroundSpecial = false
        local specialUnitsIDs = {}

        local function getCash()
            local txt = guiCash.Text:gsub("%D", "")
            return tonumber(txt) or 0
        end

        local function getMyUnits()
            local list = {}
            for _, obj in ipairs(entities:GetChildren()) do
                if obj.Name == UNIT_NAME or obj.Name == "unit_molten_flower" then
                    table.insert(list, obj)
                end
            end
            return list
        end

        local function getUnitId(unit)
            local id = unit:GetAttribute("ID")
            return id and tonumber(id) or nil
        end

        -- –°—Ç–∞–≤–∏–º —é–Ω–∏—Ç–∞ –ø–æ –∏–Ω–¥–µ–∫—Å—É –≤ –∫—Ä—É–≥–µ
        local function placeUnitAtIndex(index)
            local angle = (index / UNIT_COUNT) * math.pi * 2
            local x = math.cos(angle) * RADIUS
            local z = math.sin(angle) * RADIUS
            local args = {
                UNIT_NAME,
                {
                    Valid = true,
                    Rotation = 180,
                    CF = CFrame.new(x, 0, z, -1,0,0,0,1,0,0,0,-1),
                    Position = Vector3.new(x,0,z)
                }
            }
            local success, err = pcall(function()
                placeUnit:InvokeServer(unpack(args))
            end)
            if success then
                placedUnits += 1
                addLog(("ü™ñ –ü–æ—Å—Ç–∞–≤–ª–µ–Ω —é–Ω–∏—Ç #%d –≤ –∫—Ä—É–≥–µ"):format(placedUnits))
            else
                addLog("‚ö†Ô∏è –û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ —é–Ω–∏—Ç–∞: " .. tostring(err))
            end
        end

        -- –°—Ç–∞–≤–∏–º —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω—ã–π —Å–ø–µ—Ü. —é–Ω–∏—Ç
        local function placeCentralSpecial()
            if not placedCentral and getCash() >= 1000 then
                local args = {
                    "unit_molten_flower",
                    {
                        Valid=true,
                        Rotation=180,
                        CF=CFrame.new(0,0,0, -1,0,0, 0,1,0, 0,0,-1),
                        Position=Vector3.new(0,0,0)
                    }
                }
                local success, err = pcall(function()
                    placeUnit:InvokeServer(unpack(args))
                end)
                if success then
                    placedCentral = true
                    addLog("ü™ñ –¶–µ–Ω—Ç—Ä–∞–ª—å–Ω—ã–π —Å–ø–µ—Ü. —é–Ω–∏—Ç –ø–æ—Å—Ç–∞–≤–ª–µ–Ω")
                else
                    addLog("‚ö†Ô∏è –û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω–æ–≥–æ —Å–ø–µ—Ü. —é–Ω–∏—Ç–∞: "..tostring(err))
                end
            end
        end

                -- –°—Ç–∞–≤–∏–º 4 —Å–ø–µ—Ü. —é–Ω–∏—Ç–∞ –≤–æ–∫—Ä—É–≥ –∫—Ä—É–≥–∞
        local function placeAroundSpecial()
            if placedAroundSpecial or getCash() < 4000 then return end
            local positions = {
                Vector3.new(RADIUS+2,0,0),
                Vector3.new(0,0,RADIUS+2),
                Vector3.new(-RADIUS-2,0,0),
                Vector3.new(0,0,-RADIUS-2)
            }
            for i,pos in ipairs(positions) do
                local args = {
                    "unit_molten_flower",
                    {
                        Valid=true,
                        Rotation=180,
                        CF=CFrame.new(pos.X,pos.Y,pos.Z, -1,0,0,0,1,0,0,0,-1),
                        Position=pos
                    }
                }
                local success, err = pcall(function()
                    placeUnit:InvokeServer(unpack(args))
                end)
                if success then
                    -- –ñ–¥–µ–º, —á—Ç–æ–±—ã —Å–µ—Ä–≤–µ—Ä –¥–æ–±–∞–≤–∏–ª ID
                    task.wait(0.3)
                    -- –ò—â–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–π –ø–æ—Å—Ç–∞–≤–ª–µ–Ω–Ω—ã–π —é–Ω–∏—Ç —Å —ç—Ç–∏–º –∏–º–µ–Ω–µ–º
                    for _,unit in ipairs(entities:GetChildren()) do
                        if unit.Name == "unit_molten_flower" then
                            local uid = getUnitId(unit)
                            if uid and not table.find(specialUnitsIDs, uid) then
                                table.insert(specialUnitsIDs, uid)
                                break
                            end
                        end
                    end
                    addLog(("ü™ñ –°–ø–µ—Ü. —é–Ω–∏—Ç #%d –≤–æ–∫—Ä—É–≥ –∫—Ä—É–≥–∞ –ø–æ—Å—Ç–∞–≤–ª–µ–Ω"):format(i))
                else
                    addLog("‚ö†Ô∏è –û—à–∏–±–∫–∞ —É—Å—Ç–∞–Ω–æ–≤–∫–∏ —Å–ø–µ—Ü. —é–Ω–∏—Ç–∞: "..tostring(err))
                end
                task.wait(0.3)
            end
            placedAroundSpecial = true
        end

        -- –§—É–Ω–∫—Ü–∏—è –∞–ø–≥—Ä–µ–π–¥–∞
        local function upgradeUnitById(uid)
            while uid and unitProgress[uid] and unitProgress[uid] < #UPGRADE_COSTS do
                local cost = UPGRADE_COSTS[unitProgress[uid]+1]
                if getCash() >= cost then
                    local success, err = pcall(function()
                        upgradeUnit:InvokeServer(uid)
                    end)
                    if success then
                        unitProgress[uid] += 1
                        addLog("‚¨ÜÔ∏è –Æ–Ω–∏—Ç "..uid.." —É–ª—É—á—à–µ–Ω –¥–æ "..unitProgress[uid])
                    else
                        addLog("‚ö†Ô∏è –û—à–∏–±–∫–∞ –∞–ø–≥—Ä–µ–π–¥–∞ —é–Ω–∏—Ç–∞ "..uid..": "..tostring(err))
                        break
                    end
                    task.wait(0.5)
                else break end
            end
        end

                -- –ö–∞—á–∞–µ–º —Å–Ω–∞—á–∞–ª–∞ –∫—É–∫—É—Ä—É–∑—É
        local function upgradeCorn()
            local units = getMyUnits()
            local cornUnits = {}

            -- –°–æ–±–∏—Ä–∞–µ–º –≤—Å–µ –∫—É–∫—É—Ä—É–∑—ã
            for _,unit in ipairs(units) do
                if unit.Name == "unit_buff_corn" then
                    table.insert(cornUnits, unit)
                end
            end

            -- –ö–∞—á–∞–µ–º –∫–∞–∂–¥—É—é –∫—É–∫—É—Ä—É–∑—É –ø–æ –æ–¥–Ω–æ–π –¥–æ –º–∞–∫—Å–∏–º—É–º–∞
            for _,unit in ipairs(cornUnits) do
                local uid = getUnitId(unit)
                if uid then
                    if not unitProgress[uid] then unitProgress[uid] = 0 end
                    local level = unitProgress[uid]

                    while level < #UPGRADE_COSTS do
                        local cost = UPGRADE_COSTS[level + 1]
                        if getCash() >= cost then
                            local success, err = pcall(function()
                                upgradeUnit:InvokeServer(uid)
                            end)
                            if success then
                                level += 1
                                unitProgress[uid] = level
                                addLog(("‚¨ÜÔ∏è –ö—É–∫—É—Ä—É–∑–∞ %d ‚Üí %d (üí∞ %d)"):format(uid, level, cost))
                                task.wait(0.5)
                            else
                                addLog("‚ö†Ô∏è –û—à–∏–±–∫–∞ –∞–ø–≥—Ä–µ–π–¥–∞ –∫—É–∫—É—Ä—É–∑—ã "..uid..": "..tostring(err))
                                break
                            end
                        else
                            -- –µ—Å–ª–∏ –¥–µ–Ω–µ–≥ –Ω–µ—Ç ‚Äî –≤—ã—Ö–æ–¥–∏–º –∏–∑ —Ü–∏–∫–ª–∞
                            break
                        end
                    end
                end
            end
        end


        -- –ö–∞—á–∞–µ–º 4 —Å–ø–µ—Ü. —é–Ω–∏—Ç–∞
        local function upgradeSpecialAround()
            for _,uid in ipairs(specialUnitsIDs) do
                if not unitProgress[uid] then unitProgress[uid]=0 end
                upgradeUnitById(uid)
            end
        end

        -- –ö–∞—á–∞–µ–º –æ—Å—Ç–∞–ª—å–Ω—ã–µ —é–Ω–∏—Ç—ã
        local function upgradeOthers()
            local units = getMyUnits()
            for _,unit in ipairs(units) do
                if unit.Name ~= "unit_buff_corn" and not table.find(specialUnitsIDs,getUnitId(unit)) then
                    local uid = getUnitId(unit)
                    if uid and not unitProgress[uid] then unitProgress[uid]=0 end
                    upgradeUnitById(uid)
                end
            end
        end

        local function autoSkip()
            pcall(function()
                skipWave:InvokeServer({"y"})
            end)
        end

        local function resetCycle()
            placedUnits = 0
            unitProgress = {}
            placedCentral = false
            placedAroundSpecial = false
            specialUnitsIDs = {}
            addLog("üîÑ –ù–æ–≤—ã–π —Ü–∏–∫–ª —Ñ–∞—Ä–º–∞.")
        end

        local function getLevelFromTitle()
            local text = titleLabel.Text or ""
            return tonumber(text:match("Level%s+(%d+)"))
        end

        while farming do
            local lvl = getLevelFromTitle()
            if lvl and lvl ~= currentLevel then
                currentLevel = lvl
                resetCycle()
            end

            if not placedCentral then
                placeCentralSpecial()
            elseif placedUnits < UNIT_COUNT then
                if getCash() >= UNIT_COST then
                    placeUnitAtIndex(placedUnits+1)
                    task.wait(0.5)
                end
            else
                -- –ö–∞—á–∞–µ–º –∫—É–∫—É—Ä—É–∑—É
                upgradeCorn()
                -- –°—Ç–∞–≤–∏–º 4 —Å–ø–µ—Ü. —é–Ω–∏—Ç–∞
                placeAroundSpecial()
                upgradeSpecialAround()
                -- –ê–ø–≥—Ä–µ–π–¥ –æ—Å—Ç–∞–ª—å–Ω—ã—Ö
                upgradeOthers()
            end

            if screenMiddle:FindFirstChild("DifficultyVote") and screenMiddle.DifficultyVote.Visible then
                voteDifficulty:InvokeServer("dif_impossible")
                addLog("üó≥Ô∏è –ü—Ä–æ–≥–æ–ª–æ—Å–æ–≤–∞–ª –∑–∞ dif_impossible.")
                task.wait(1)
                continue
            end

            if screenMiddle:FindFirstChild("GameEnd") and screenMiddle.GameEnd.Visible then
                restartGame:InvokeServer()
                addLog("üîÅ –ü–µ—Ä–µ–∑–∞–ø—É—Å–∫ –∏–≥—Ä—ã.")
                resetCycle()
                task.wait(2)
                continue
            end

            autoSkip()
            task.wait(1)
        end
    end)
end)

local function createAntiAFKGui()
    antiAFKGui = Instance.new("ScreenGui")
    antiAFKGui.Name = "AntiAFK_UI"
    antiAFKGui.ResetOnSpawn = false
    antiAFKGui.Parent = player:WaitForChild("PlayerGui")

    local frame = Instance.new("Frame", antiAFKGui)
    frame.Size = UDim2.new(0, 280, 0, 100)
    frame.Position = UDim2.new(0, 10, 0, 10)
    frame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    frame.BorderSizePixel = 0
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 10)

    local title = Instance.new("TextLabel", frame)
    title.Size = UDim2.new(1, 0, 0, 30)
    title.BackgroundTransparency = 1
    title.Font = Enum.Font.GothamBold
    title.TextSize = 20
    title.TextColor3 = Color3.fromRGB(200, 200, 200)
    title.Text = "Anti AFK"

    local toggleBtn = Instance.new("TextButton", frame)
    toggleBtn.Size = UDim2.new(0, 120, 0, 40)
    toggleBtn.Position = UDim2.new(0, 10, 0, 50)
    toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 120, 50)
    toggleBtn.Font = Enum.Font.GothamBold
    toggleBtn.TextSize = 18
    toggleBtn.TextColor3 = Color3.fromRGB(230, 230, 230)
    toggleBtn.Text = "OFF"
    Instance.new("UICorner", toggleBtn).CornerRadius = UDim.new(0, 6)

    local timerLabel = Instance.new("TextLabel", frame)
    timerLabel.Size = UDim2.new(0, 130, 0, 40)
    timerLabel.Position = UDim2.new(0, 150, 0, 50)
    timerLabel.BackgroundTransparency = 1
    timerLabel.Font = Enum.Font.GothamBold
    timerLabel.TextSize = 18
    timerLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    timerLabel.TextXAlignment = Enum.TextXAlignment.Left
    timerLabel.Text = "Idle Time: 0s"

    local toggleEnabled = false
    local localIdleTime = 0
    local lastResetTime = 0

    local function resetIdle()
        VirtualUser:CaptureController()
        VirtualUser:ClickButton2()
    end

    local function onInput()
        localIdleTime = 0
        lastResetTime = 0
    end

    toggleBtn.MouseButton1Click:Connect(function()
        toggleEnabled = not toggleEnabled
        if toggleEnabled then
            toggleBtn.Text = "ON"
            toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 180, 50)
            localIdleTime = 0
            lastResetTime = 0
        else
            toggleBtn.Text = "OFF"
            toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 120, 50)
            localIdleTime = 0
            lastResetTime = 0
        end
    end)

    UserInputService.InputBegan:Connect(onInput)
    UserInputService.InputChanged:Connect(onInput)

    RunService.Heartbeat:Connect(function(dt)
        if toggleEnabled then
            localIdleTime = localIdleTime + dt
            timerLabel.Text = ("Idle Time: %.0fs"):format(localIdleTime)

            if localIdleTime - lastResetTime >= 15 then
                resetIdle()
                lastResetTime = localIdleTime
                VirtualUser:CaptureController()
                VirtualUser:MoveMouseRel(math.random(-1, 1), math.random(-1, 1))
            end
        else
            localIdleTime = 0
            lastResetTime = 0
            timerLabel.Text = "Idle Time: 0s"
        end
    end)
end


antiAFKBtn.MouseButton1Click:Connect(function()
    if not antiAFKGui then
        createAntiAFKGui()
        enabled = true
		addLog("‚úÖ antiAfk —É—Å–ø–µ—à–Ω–æ –∑–∞–≥—Ä—É–∂–µ–Ω.")
    else
        enabled = not enabled
        antiAFKGui.Enabled = enabled
    end
end)
